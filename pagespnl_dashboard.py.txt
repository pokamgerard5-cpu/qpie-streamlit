# Initialize PnL tracking if not exists

# Function to simulate trade (will integrate with real predictions later)
def add_trade(outcome_predicted, outcome_actual, confidence, bet_size_usd):
    if confidence < 0.6:
        # Skip low confidence
        pnl = 0
        trade_result = "SKIPPED"
    elif outcome_predicted == outcome_actual:
        # Win
        if outcome_actual == "TIE":
            pnl = bet_size_usd * 2.5  # Ties pay higher
        else:
            pnl = bet_size_usd * 0.95  # Standard win (after edge)
        trade_result = "WIN"
    else:
        # Loss
        pnl = -bet_size_usd
        trade_result = "LOSS"
    
    st.session_state.cumulative_pnl += pnl
    
    trade = {
        "timestamp": datetime.now(),
        "predicted": outcome_predicted,
        "actual": outcome_actual,
        "confidence": confidence,
        "bet_size": bet_size_usd,
        "pnl": pnl,
        "cumulative": st.session_state.cumulative_pnl,
        "result": trade_result
    }
    st.session_state.trades.append(trade)
    
    # Keep only last 1000 trades for performance
    if len(st.session_state.trades) > 1000:
        st.session_state.trades = st.session_state.trades[-1000:]
    
    return pnl
